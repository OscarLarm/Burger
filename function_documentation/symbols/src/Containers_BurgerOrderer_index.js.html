 38 /**
 39  * Returns a html-formatted string stating that an order has been placed for the values from 
 40  * the keys "burger" and "ingredients" in argument data.
 41  * @public
 42  * @param {*} data The data being read.
 43  * @returns {string} String formatted as html, stating the orders placed from data.
 44  */
 45 function orderPage(data){
 46     JSON.stringify(data)
 47     page = '<h2>Order sent!</h2>'
 48     page += `<a href = "/">Back</a>`
 49     page += `<hr>`
 50     page += `<p>One ${data["foodItem"]}`
 51     if (data["ingredients"]) {
 52         page += ` with `
 53         let ingredients = [data["ingredients"]]
 54         ingredients.join(", ")
 55         page += `${ingredients}.`
 56     }
 57     page += `</p>`
 58     return page
 59 };
