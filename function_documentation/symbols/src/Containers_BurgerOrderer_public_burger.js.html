1 /**
  2  * Fetches all data from the /database route with GET.
  3  * @function
  4  * @async
  5  * @returns Data parsed from json to js object.
  6  */
  7 const getData = async () => {
  8     var res = await fetch('/database');
  9     var menu = await res.json()
 10     return menu
 11 }
 12 /**
 13  * Used to generate the forms that send order data from the burgerOrderer site.
 14  * Reads all data from the getData() function, then creates a new array containing only the objects with the key-value of "type": "argument".
 15  * It then create a HTML form with input fields for the key "name" and "ingredients" if the values exist.
 16  * @function
 17  * @param {string} food - The value from the key "type".
 18  * @async
 19  * @example
 20  * renderFood('burger')
 21  * <form action="/order" method="post" id="Original Chicken Burger">
 22  * <h2>Original Chicken Burger</h2>
 23  * <input type="hidden" name="foodItem" value="Original Chicken Burger"></input>
 24  * <h4>Ingredients:</h4>
 25  * <input type="checkbox" name="ingredients" value="Fried Chicken Patty">Fried Chicken Patty
 26  * <input type="checkbox" name="ingredients" value="Bread">Bread
 27  * <input type="submit" value="Buy">
 28  * </form>
 29  */
 30 const renderFood = async(food) => {
 31     const menuData = await getData()
 32     const foodItem = menuData.filter(item => item.type === food);
 33 
 34     var container = document.getElementById("container")
 35     container.innerHTML += `<ul id=foodList>`
 36     var foodList = document.getElementById("foodList")
 37     foodItem.forEach(item => {
 38         foodList.innerHTML += `<form action="/order" method="post" id="${item.name}">`
 39         container = document.getElementById(`${item.name}`)
 40         container.innerHTML += `<h2>${item.name}</h2>`
 41         container.innerHTML += `<input type="hidden" name="foodItem" value="${item.name}"</input>`
 42         if (item.ingredients.length !== 0){
 43             container.innerHTML += `<h4>Ingredients:</h4>`;
 44             item["ingredients"].forEach(ingredient => {
 45                 container.innerHTML += `<input type="checkbox" name ="ingredients" value = "${ingredient}">${ingredient}</input>`;
 46             });
 47         }
 48         container.innerHTML += `<input type="submit" value= "Buy"></input>`
 49         container.innerHTML += `</form>`
 50     });
 51     container.innerHTML += "</ul>";
 52 
 53     document.querySelectorAll(".burgerCheckbox").forEach(burgerCheckbox => {
 54         burgerCheckbox.addEventListener("change", function () {
 55             const form = this.closest("form");
 56             const ingredientsCheckboxes = form.querySelectorAll(".ingredientCheckbox");
 57 
 58             ingredientsCheckboxes.forEach(checkbox => {
 59                 checkbox.checked = this.checked
 60             });
 61         });
 62     });
 63 }
 64 
 65 renderFood('burger');
 66 renderFood('salad');
 67 renderFood('sides');
 68 renderFood('drinks');
